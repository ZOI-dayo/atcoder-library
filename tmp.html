1 #pragma once
 2 #include "../common/template.hpp"
 3 #include "point.hpp"
 4 
 5 class Area {
 6 public:
 7   int H, W;
 8   bool allow_outside = false;
 9   char out_val;
10   Area(int H, int W) : H(H), W(W), data(vector(H, string(W, '.'))) {}
11   Area(int H, int W, char out_val)
12       : H(H), W(W), data(vector(H, string(W, '.'))), allow_outside(true),
13         out_val(out_val) {}
14   string &operator[](int i) { return data[i]; }
15   // 時計回りで90度回転
16   Area rotated90() const {
17     Area ret(W, H);
18     rep(i, H) rep(j, W) ret[j][H - i - 1] = data[i][j];
19     return ret;
20   }
21   void rotate90() {
22     assert(H == W);
23     data = rotated90().data;
24   }
25   char at(int x, int y) {
26     if (is_contained(H, W, x, y))
27       return data[x][y];
28     else if (allow_outside)
29       return out_val;
30     else {
31       cerr << "[Area::at] out of range" << endl;
32       assert(false);
33     }
34   }
35   void set(int x, int y, char val) {
36     if (is_contained(H, W, x, y))
37       data[x][y] = val;
38     else if (allow_outside)
39       return;
40     else {
41       cerr << "[Area::set] out of range" << endl;
42       assert(false);
43     }
44   }
45 
46   bool contains(int x, int y) const { return is_contained(H, W, x, y); }
47   bool contains(Point p) const { return is_contained(H, W, p.x, p.y); }
48 
49 private:
50   vec<string> data;
51 };
52 ostream &operator<<(ostream &os, Area &a) {
53   rep(i, a.H) os << a[i] << endl;
54   return os;
55 }
56 istream &operator>>(istream &is, Area &a) {
57   rep(i, a.H) is >> a[i];
58   return is;
59 }
